def make_meshgrid(x, y, h=.02):
    x_min, x_max = x.min() - 1, x.max() + 1
    y_min, y_max = y.min() - 1, y.max() + 1
    xx, yy = np.meshgrid(np.arange(x_min, x_max, h),
                         np.arange(y_min, y_max, h))
    return xx, yy


def plot_contours(ax, clf, xx, yy, **params):
    Z = clf.predict(np.c_[xx.ravel(), yy.ravel()])
    Z = Z.reshape(xx.shape)
    out = ax.contourf(xx, yy, Z, **params)
    return out

def fit_and_measure_time(clf, X_train, y_train):
    import time
    start = time.time()
    clf.fit(X_train, y_train)
    end = time.time()
    return end - start

def get_training_time_and_acc(clf, X_train, y_train, X_val, y_val):
    return fit_and_measure_time(clf, X_train, y_train), clf.score(X_val, y_val)
    
def compare_svm_and_random_forest(svm, rf, X_train, y_train, X_val, y_val):
    svm_time, svm_score = get_training_time_and_acc(svm, X_train, y_train, X_val, y_val)
    rf_time, rf_score = get_training_time_and_acc(rf, X_train, y_train, X_val, y_val)
    print("===SVM===")
    print("Time: " + str(svm_time))
    print("Accuracy: " + str(svm_score))
    print("===Random Forest===")
    print("Time: " + str(rf_time))
    print("Accuracy: " + str(rf_score))
    print("===(SVM - Random Forest)===")
    print("Time: " + str(svm_time - rf_time))
    print("Accuracy: " + str(svm_score - rf_score))

def get_class_proba(y):
    _, counts = np.unique(y, return_counts=True)
    return counts / y.shape[0]

def create_labels_mapping(y):
    classes, inverse = np.unique(y, return_inverse=True)
    real_labels_to_int = {}
    for i in range(classes.shape[0]):
        real_labels_to_int[classes[i]] = i
    return real_labels_to_int, classes, inverse

class MyGaussianNB:
    def calculate_mean_and_std(self, X, y):
        self.features_means = np.empty((self.int_to_real_labels.shape[0], X.shape[1]))
        self.features_stds = np.empty((self.int_to_real_labels.shape[0], X.shape[1]))
        for label in range(self.int_to_real_labels.shape[0]):
            features_for_label = X[y == label, :]
            self.features_means[label] = np.mean(features_for_label, axis=0)
            self.features_stds[label] = np.std(features_for_label, axis=0)
    
    def __init__(self):
        self.class_proba = None
        self.features_proba = None
        self.real_labels_to_int = None
        self.int_to_real_labels = None
        self.features_means = None
        self.features_stds = None
    
    def fit(self, X, y):
        self.real_labels_to_int, self.int_to_real_labels, y = create_labels_mapping(y)
        self.class_proba = get_class_proba(y)
        self.calculate_mean_and_std(X, y)
        
    def predict_proba(self, X):
        proba = np.empty((X.shape[0], self.int_to_real_labels.shape[0]))
        for label in range(self.int_to_real_labels.shape[0]):
            pdfs = np.empty(X.shape)
            for i in range(X.shape[0]):
                for j in range(X.shape[1]):
                    pdfs[i][j] = norm.pdf(X[i][j], loc=self.features_means[label][j], scale=self.features_stds[label][j])
            proba[:, label] = np.prod(pdfs, axis=1) * self.class_proba[label]
        for i in range(proba.shape[0]):
            if np.sum(proba[i]) != 0:
                proba[i] = proba[i] / np.sum(proba[i])
            else:
                proba[i] = 0.5
        return proba
    
    def convert_labels_to_int(self, y):
        return np.array([self.real_labels_to_int[y[i]] for i in range(y.shape[0])])
    
    def convert_int_to_real_labels(self, y):
        return np.array([self.int_to_real_labels[y[i]] for i in range(y.shape[0])])
    
    def predict(self, X):
        proba = self.predict_proba(X)
        predicted = np.argmax(proba, axis=1)
        return self.convert_int_to_real_labels(predicted)
    
    def score(self, X, y):
        predicted = self.predict(X)
        return (predicted == y).mean()

def split_string_on_words(string):
    ps = PorterStemmer() 
    words = word_tokenize(string)
    stemmed_words = [ps.stem(word) for word in words]
    return stemmed_words

def convert_texts_to_words_list(texts):
    return np.array([word for i in range(texts.shape[0]) for word in split_string_on_words(texts[i])])

def convert_texts_to_words_lists_for_each(texts):
    return [split_string_on_words(texts[i]) for i in range(texts.shape[0])]

def transform_text_to_frequency_map(texts, y):
    text_words = convert_texts_to_words_list(texts)
    unique_words, counts = np.unique(text_words, return_counts=True)
    most_frequent = np.argsort(counts)[-int(1 * len(counts)):]
    unique_words = unique_words[most_frequent]
    classes = np.unique(y)
    counts_for_label = np.zeros((classes.shape[0], unique_words.shape[0]))
    for i in range(counts_for_label.shape[0]):
        words_for_class = convert_texts_to_words_list(texts[y == i])
        for j in range(unique_words.shape[0]):
            counts_for_label[i][j] += np.sum(words_for_class == unique_words[j])
    ids_to_words = {unique_words[i]: i for i in range(unique_words.shape[0])}
    return ids_to_words, counts_for_label

class MyMultinomialNB:
    def fit(self, X, y):
        word_mapping, word_frequency_per_class = transform_text_to_frequency_map(X, y)
        print("Vocabulary size: " + str(len(word_mapping)))
        self.class_proba = get_class_proba(np.arange(word_frequency_per_class.shape[0]))
        self.word_frequency_per_class = word_frequency_per_class
        self.word_mapping = word_mapping
        
    def predict_proba(self, X):
        X = convert_texts_to_words_lists_for_each(X)
        proba = np.empty((len(X), self.class_proba.shape[0]))
        for label in range(self.class_proba.shape[0]):
            for i in range(len(X)):
                pdf = 1
                for j in range(len(X[i])):
                    if X[i][j] in self.word_mapping:
                        word_id = self.word_mapping[X[i][j]]
                        pdf *= (self.word_frequency_per_class[label][word_id] + 1) / (np.sum(self.word_frequency_per_class[label]) + len(self.word_mapping))
                proba[i, label] = pdf * self.class_proba[label]
        for i in range(proba.shape[0]):
            if np.sum(proba[i]) != 0:
                proba[i] = proba[i] / np.sum(proba[i])
            else:
                proba[i] = 0.5
        return proba
    
    def predict(self, X):
        proba = self.predict_proba(X)
        predicted = np.argmax(proba, axis=1)
        return predicted
    
    def score(self, X, y):
        predicted = self.predict(X)
        return (predicted == y).mean()
